library(ggrepel)
mlbf <- read_csv("~/Library/CloudStorage/OneDrive-UniversityofNebraska-Lincoln/Fall 2022/SPMC350/SPMC350 Blog/GRAPHS and DATA/p1 data/fielding22.csv")

  filter(Tm== "Philadelphia Phillies")

ggplot()+
  geom_point(

    aes(
    x= DefEff,
    y= OPS),
  color= "gray") +

    data= combinedphi,
    aes(
    x= DefEff,
    y= OPS),
  color= "#E81828")+

  postphillies %>%
